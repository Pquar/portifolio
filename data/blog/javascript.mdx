### Arrays

```jsx
//Criar um Array= Lista
let minhaArray = [];

//Enviando elemento para o Array
minhaArray.push(1)
minhaArray.push(2)

// Percorendo o array
//for of - pegando os elementos a cada intereção
for(let elemento of arr) {
		minhaArray[elemento]
}
//for in - percore os indices do elementos= 0,1,2,3

for(let elemento in minhaArray){
			minhaArray[elemento]
}
//
```

```jsx
//Transformar em texto
```

# Function

```jsx

function nomeFuncao() {
		console.log('voce esta dentro da funcao')
}

//chama a fução
nomeFuncao()

//Definindo uma função com parametro
function nomeFuncaoParametro(nome) {
		console.log(nome +' esta dentro da funcao')
}
function parametros(nome, idade, altura='1.7') {
		console.log(nome + idade +  altura' esta dentro da funcao')
}

//chama a fução
nomeFuncaoParametro('nome')
nomeFuncaoParametro() // undefined

//retorno de função

			return numero1 + numero1 // a função acaba quando chega no return

```

# Objetos

```jsx
let pessoa = {
		nome: 'nome pessoa',
		'idade-da-pessoa': 23, 
}

pessoa.nome
pessoa.['nome']
pessoa.["idade-da-pessoa"]

//adicionar
pessoa.altura = 1.7

console.log(pessoa.altura)

//remover

delete pessoa.altura

//percorer o objeto
//for in

for (let chave in pessoa){
		console.log(chave)
}
```
